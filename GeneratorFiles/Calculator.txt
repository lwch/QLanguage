%token "+" "-" "*" "/" "(" ")" "sin" "cos" "log";

%start factor;

factor -> "{Digit}"
       |  "+" factor
       |  "-" factor
       |  "(" exp ")"
       |  "sin" "(" exp ")"
       |  "cos" "(" exp ")"
       |  "log" "(" exp ")"
       ;

term -> term "*" factor
     |  term "/" factor
     |  factor
     ;

exp -> exp "+" term
    |  exp "-" term
    ;
