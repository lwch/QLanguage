%token "+" "-" "*" "/" "(" ")" "sin" "cos" "log";

%start sym;

value_type -> "{Real}"
           |  "{Digit}"
           ;

sym -> "+" exp
    |  "-" exp
    |  exp
    ;

exp -> exp "+" term
    |  exp "-" term
    |  term
    ;

term -> term "*" factor
     |  term "/" factor
     |  factor
     ;

factor -> "sin" bracket
       |  "cos" bracket
       |  "log" bracket
       |  bracket
       |  value_type
       ;

bracket -> "(" sym ")"
        ;
