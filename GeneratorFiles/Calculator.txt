%token "+" "-" "*" "/" "(" ")" "sin" "cos" "log";

%start exp;

value_type -> "{Real}"
           |  "{Digit}"
		   ;

exp -> exp "+" term
    |  exp "-" term
	|  term
    ;

term -> term "*" factor
     |  term "/" factor
     |  factor
     ;

factor -> "(" exp ")"
       |  "sin" "(" exp ")"
       |  "cos" "(" exp ")"
       |  "log" "(" exp ")"
	   |  "+" value_type
	   |  "-" value_type
	   |  value_type
	   ;
